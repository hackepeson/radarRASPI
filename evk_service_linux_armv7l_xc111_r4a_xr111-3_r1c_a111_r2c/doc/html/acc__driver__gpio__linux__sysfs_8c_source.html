71 /**
72  * @brief Internal GPIO open
73  *
74  * Export GPIO to create /sys/class/gpio/gpio#
75  * Create /sys/class/gpio/gpio#/value and /sys/class/gpio/gpio#/direction.
76  *
77  * @param pin GPIO pin
78  * @return Status
79  */
80 static acc_status_t internal_gpio_open(uint_fast8_t pin)
81 {
82  int unexport_fd;
83  int export_fd;
84  char gpio_x[5];
85  ssize_t gpio_x_len;
86  ssize_t bytes_written;
87  char dir_path[sizeof(GPIO_DIRECTION_PATH) + 10];
88  char value_path[sizeof(GPIO_VALUE_PATH) + 10];
89 
90  if (pin >= gpio_count) {
91  ACC_LOG_ERROR("GPIO %u is not a valid GPIO pin", pin);
92  return ACC_STATUS_BAD_PARAM;
93  }
94  gpio_t *gpio = &gpios[pin];
95 
96  if (gpio->is_open)
97  return ACC_STATUS_SUCCESS;
98 
99  gpio_x_len = snprintf(gpio_x, sizeof(gpio_x), "%u", pin);
100  snprintf(dir_path, sizeof(dir_path), GPIO_DIRECTION_PATH, pin);
101  snprintf(value_path, sizeof(value_path), GPIO_VALUE_PATH, pin);
102 
103  // Clean-up of gpio
104  unexport_fd = open(GPIO_UNEXPORT_PATH, O_WRONLY);
105  bytes_written = write(unexport_fd, gpio_x, gpio_x_len);
106  close(unexport_fd);
107 
108  export_fd = open(GPIO_EXPORT_PATH, O_WRONLY);
109  if (export_fd < 0) {
110  ACC_LOG_FATAL("Unable to open gpio export: %s", strerror(errno));
111  return ACC_STATUS_FAILURE;
112  }
113  bytes_written = write(export_fd, gpio_x, gpio_x_len);
114 
115  if (bytes_written < 0) {
116  ACC_LOG_ERROR("Could not write to gpio export: %s", strerror(errno));
117  close(export_fd);
118  return ACC_STATUS_FAILURE;
119  }
120 
121  if (bytes_written != gpio_x_len) {
122  ACC_LOG_ERROR("Expected to write %d bytes to gpio export, but wrote: %d", (int)gpio_x_len, (int)bytes_written);
123  close(export_fd);
124  return ACC_STATUS_FAILURE;
125  }
126 
127  close(export_fd);
128 
129  gpio->dir_fd = -1;
130  gpio->value_fd = -1;
131  gpio->dir = GPIO_DIR_UNKNOWN;
132  gpio->value = 0;
133  gpio->pull = 0;
134 
135  // Wait a maximum of one second for each gpio to open
136  for (uint_fast16_t loop_count = 0; loop_count < 100; loop_count++) {
137  gpio->dir_fd = open(dir_path, O_RDWR);
138  if (gpio->dir_fd < 0) {
139  acc_os_sleep_us(10000);
140  }
141  else {
142  ACC_LOG_VERBOSE("Waited %u ms on gpio open direction", loop_count * 10);
143  break;
144  }
145  }
146 
147  if (gpio->dir_fd < 0) {
148  ACC_LOG_ERROR("Unable to open gpio%u direction: %s", pin, strerror(errno));
149  return ACC_STATUS_FAILURE;
150  }
151 
152  // Wait a maximum of one second for each gpio to open
153  for (uint_fast16_t loop_count = 0; loop_count < 100; loop_count++) {
154  gpio->value_fd = open(value_path, O_RDWR);
155  if (gpio->value_fd < 0) {
156  acc_os_sleep_us(10000);
157  }
158  else {
159  ACC_LOG_VERBOSE("Waited %u ms on gpio open value", loop_count * 10);
160  break;
161  }
162  }
163 
164  if (gpio->value_fd < 0) {
165  ACC_LOG_ERROR("Unable to open gpio%u value: %s", pin, strerror(errno));
166  return ACC_STATUS_FAILURE;
167  }
168 
169  gpio->is_open = 1;
170 
171  return ACC_STATUS_SUCCESS;
172 }

175 /**
176  * @brief Internal GPIO set direction
177  *
178  * If 'dir' is GPIO_DIR_IN, 'value' is not used.
179  *
180  * @param gpio GPIO information
181  * @param dir The direction to be set
182  * @param value The value to be written
183  * @return Status
184  */
185 static acc_status_t internal_gpio_set_dir(gpio_t *gpio, gpio_dir_t dir, uint_fast8_t value)
186 {
187  ssize_t bytes_written;
188  char *dir_str = (dir == GPIO_DIR_IN) ? "in" : ((value == 0) ? "low" : "high");
189  size_t dir_str_len = strlen(dir_str);
190 
191  bytes_written = write(gpio->dir_fd, dir_str, dir_str_len);
192  if (bytes_written < 0) {
193  ACC_LOG_ERROR("Could not write to gpio%u direction: %s", gpio->pin, strerror(errno));
194  return ACC_STATUS_FAILURE;
195  } else if (bytes_written != (ssize_t)dir_str_len) {
196  ACC_LOG_ERROR("Expected to write %u bytes to GPIO direction, but wrote: %d bytes", (unsigned int)dir_str_len, (int)bytes_written);
197  return ACC_STATUS_FAILURE;
198  }
199  gpio->dir = dir;
200  gpio->value = value;
201 
202  return ACC_STATUS_SUCCESS;
203 }

206 /**
207  * @brief Internal GPIO write
208  *
209  * Write 'value' to /sys/class/gpio/gpio#/value.
210  * GPIO needs to already be an output.
211  *
212  * @param gpio GPIO information
213  * @param value The value to be written
214  * @return Status
215  */
216 static acc_status_t internal_gpio_set_value(gpio_t *gpio, uint_fast8_t value)
217 {
218  ssize_t bytes_written;
219 
220  if (value > 1)
221  value = 1;
222  if (gpio->value == value)
223  return ACC_STATUS_SUCCESS;
224 
225  bytes_written = write(gpio->value_fd, (value == 0) ? "0" : "1", 1);
226  if (bytes_written < 0) {
227  ACC_LOG_ERROR("Could not write to gpio%u value: %s", gpio->pin, strerror(errno));
228  return ACC_STATUS_FAILURE;
229  } else if (bytes_written != 1) {
230  ACC_LOG_ERROR("Bytes written to gpio%u were not 1", gpio->pin);
231  return ACC_STATUS_FAILURE;
232  }
233 
234  gpio->value = value;
235  return ACC_STATUS_SUCCESS;
236 }

239 /**
240  * @brief Internal GPIO close all gpios
241  *
242  * Unexport GPIO to remove /sys/class/gpio/gpio#.
243  */
244 static void internal_gpio_close_all(void)
245 {
246  int unexport_fd;
247  char gpio_x[5];
248  ssize_t gpio_x_len;
249  ssize_t bytes_written;
250  gpio_t *gpio;
251 
252  unexport_fd = open(GPIO_UNEXPORT_PATH, O_WRONLY);
253  if (unexport_fd < 0) {
254  ACC_LOG_ERROR("Unable to open gpio unexport: %s", strerror(errno));
255  return;
256  }
257 
258  gpio = gpios;
259  for (uint_fast8_t pin = 0; pin < gpio_count; pin++, gpio++) {
260  if (gpio->is_open) {
261  if (gpio->dir == GPIO_DIR_OUT) {
262  internal_gpio_set_value(gpio, gpio->pull);
263  internal_gpio_set_dir(gpio, GPIO_DIR_IN, 0);
264  }
265 
266  close(gpio->dir_fd);
267  gpio->dir_fd = -1;
268 
269  close(gpio->value_fd);
270  gpio->value_fd = -1;
271 
272  gpio_x_len = snprintf(gpio_x, sizeof(gpio_x), "%u", gpio->pin);
273 
274  bytes_written = write(unexport_fd, gpio_x, gpio_x_len);
275  if (bytes_written < 0) {
276  ACC_LOG_ERROR("Could not write to gpio unexport for gpio%u: %s", gpio->pin, strerror(errno));
277  } else if (bytes_written != gpio_x_len) {
278  ACC_LOG_ERROR("Expected to write %d bytes to gpio unexport, but wrote %d for gpio%u", (int)gpio_x_len, (int)bytes_written, gpio->pin);
279  }
280 
281  gpio->is_open = 0;
282  }
283  }
284 
285  // Forbidden due to race condition between gpio_close_all and gpio_input/output.
286 // acc_os_mem_free(gpios);
287 
288  close(unexport_fd);
289 }

292 /**
293  * @brief Initialize GPIO driver
294  *
295  * @return Status
296  */
297 static acc_status_t acc_driver_gpio_linux_sysfs_init(void)
298 {
299  static acc_os_mutex_t init_mutex;
300  static bool init_done;
301 
302  if (init_done)
303  return ACC_STATUS_SUCCESS;
304 
305  acc_os_init();
306  acc_os_mutex_init(&init_mutex);
307 
308  acc_os_mutex_lock(&init_mutex);
309  if (init_done) {
310  acc_os_mutex_unlock(&init_mutex);
311  return ACC_STATUS_SUCCESS;
312  }
313 
314  gpios = acc_os_mem_alloc(sizeof(gpio_t) * gpio_count);
315  if (!gpios) {
316  ACC_LOG_ERROR("Out of memory");
317  acc_os_mutex_unlock(&init_mutex);
318  return ACC_STATUS_OUT_OF_MEMORY;
319  }
320 
321  memset(gpios, 0, sizeof(gpio_t) * gpio_count);
322  for (uint_fast8_t pin = 0; pin < gpio_count; pin++) {
323  gpios[pin].pin = pin;
324  gpios[pin].dir_fd = -1;
325  gpios[pin].value_fd = -1;
326  }
327 
328  if (atexit(internal_gpio_close_all)) {
329  ACC_LOG_ERROR("Unable to set exit function 'internal_gpio_close_all()'");
330  internal_gpio_close_all();
331  acc_os_mutex_unlock(&init_mutex);
332  return ACC_STATUS_FAILURE;
333  }
334 
335  init_done = true;
336  acc_os_mutex_unlock(&init_mutex);
337 
338  return ACC_STATUS_SUCCESS;
339 }

371 /**
372  * @brief Set GPIO to input
373  *
374  * This function sets the direction of a GPIO to input.
375  * GPIO parameter is not a pin number but GPIO index (0-X).
376  *
377  * @param pin GPIO pin to be set to input
378  * @return Status
379  */
380 acc_status_t acc_driver_gpio_linux_sysfs_input(uint_fast8_t pin)
381 {
382  acc_status_t status;
383  gpio_t *gpio;
384 
385  status = internal_gpio_open(pin);
386  if (status != ACC_STATUS_SUCCESS) {
387  return status;
388  }
389  gpio = &gpios[pin];
390 
391  if (gpio->dir == GPIO_DIR_IN)
392  return ACC_STATUS_SUCCESS;
393 
394  if (gpio->dir == GPIO_DIR_OUT) {
395  // Needed to prevent glitches in Raspberry Pi when changing back to output
396  status = internal_gpio_set_value(gpio, gpio->pull);
397  if (status != ACC_STATUS_SUCCESS) {
398  return status;
399  }
400  }
401 
402  status = internal_gpio_set_dir(gpio, GPIO_DIR_IN, 0);
403  if (status != ACC_STATUS_SUCCESS) {
404  return status;
405  }
406 
407  return ACC_STATUS_SUCCESS;
408 }

413  *
414  * @param pin GPIO pin to read
415  * @param value The value which has been read
416  * @return Status

418 acc_status_t acc_driver_gpio_linux_sysfs_read(uint_fast8_t pin, uint_fast8_t *value)
419 {
420  acc_status_t status;
421  gpio_t *gpio;
422 
423  status = internal_gpio_open(pin);
424  if (status != ACC_STATUS_SUCCESS) {
425  return status;
426  }
427  gpio = &gpios[pin];
428 
429  if (gpio->dir != GPIO_DIR_IN) {
430  ACC_LOG_ERROR("Cannot read GPIO %u as it is output/unknown", pin);
431  return ACC_STATUS_FAILURE;
432  }
433 
434  char value_str[10];
435  ssize_t bytes_read;
436 
437  // position file pointer at beginning of value file
438  off_t offset = lseek(gpio->value_fd, 0, SEEK_SET);
439  if (offset < 0) {
440  ACC_LOG_ERROR("Unable to lseek() GPIO %u: (%u) %s", pin, errno, strerror(errno));
441  return ACC_STATUS_FAILURE;
442  }
443  if (offset > 0) {
444  ACC_LOG_ERROR("lseek() GPIO %u returned %u", pin, offset);
445  return ACC_STATUS_FAILURE;
446  }
447 
448  // read GPIO input value
449  bytes_read = read(gpio->value_fd, value_str, sizeof(value_str));
450  if (bytes_read < 0) {
451  ACC_LOG_ERROR("Unable to read from GPIO %u: (%u) %s", pin, errno, strerror(errno));
452  return ACC_STATUS_FAILURE;
453  }
454  if (bytes_read == 0) {
455  ACC_LOG_ERROR("Zero bytes read for GPIO %u", pin);
456  return ACC_STATUS_FAILURE;
457  }
458 
459  *value = (value_str[0] != '0') ? 1 : 0;
460 
461  return ACC_STATUS_SUCCESS;
462 }
