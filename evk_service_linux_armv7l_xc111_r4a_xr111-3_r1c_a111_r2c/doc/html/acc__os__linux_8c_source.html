118 /**
119  * @brief Prepare stack for measuring stack usage - to be called as early as possible
120  *
121  * @param stack_size Amount of stack in bytes that is allocated
122  */
123 void acc_os_stack_setup(size_t stack_size)
124 {
125  if (!stack_size)
126  return;
127 
128  uint8_t stack_filler[stack_size];
129  memset(stack_filler, 0x5a, sizeof(stack_filler));
130 
131  /* Prevent compiler from optimizing away stack_filler[] */
132  __asm__ __volatile__("" :: "m" (stack_filler));
133 
134  acc_os_stack_setup_done = 1;
135 }

138 /**
139  * @brief Measure amount of used stack in bytes
140  *
141  * @param stack_size Amount of stack in bytes that is allocated
142  * @return Number of bytes of used stack space
143  */
144 size_t acc_os_stack_get_usage(size_t stack_size)
145 {
146  if (!stack_size || !acc_os_stack_setup_done)
147  return 0;
148 
149  uint8_t stack_filler[stack_size];
150  size_t usage = 0;
151 
152  /* This does not give an exact figure but is useful as an indication of the used size.
153  The reason for counting backwards to allow future improvement handling any misalignment
154  garbage at the end. */
155  for (size_t index = stack_size - 1; index; index--)
156  if (stack_filler[index] != 0x5a)
157  usage += 1;
158 
159  return usage;
160 }

489 acc_os_socket_t acc_os_net_connect(acc_os_net_address_t address, acc_os_net_port_t port)
490 {
491  acc_os_socket_t sock = acc_os_mem_alloc(sizeof(*sock));
492 
493  if (sock == NULL) {
494  return NULL;
495  }
496 
497  int sock_flags;
498 
499  if ((sock->handle = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
500  ACC_LOG_ERROR("%s: socket(AF_INET, SOCK_STREAM): (%u) %s", __func__, errno, strerror(errno));
501  return NULL;
502  }
503 
504  // make socket nonblocking to allow timeout handling for connect()
505  sock_flags = fcntl(sock->handle, F_GETFL);
506  fcntl(sock->handle, F_SETFL, sock_flags | O_NONBLOCK);
507 
508  // request connect, but don't block
509  struct sockaddr_in addr;
510 
511  memset(&addr, 0, sizeof(addr));
512  addr.sin_family = AF_INET;
513  addr.sin_addr.s_addr = address;
514  addr.sin_port = htons(port);
515 
516  if ((connect(sock->handle, (struct sockaddr*)&addr, sizeof(addr)) < 0) && (errno != EINPROGRESS)) {
517  char name[INET_ADDRSTRLEN];
518  acc_os_net_address_to_string(address, name, sizeof(name));
519  ACC_LOG_WARNING("%s: connect(%s:%u): (%u) %s", __func__, name, port, errno, strerror(errno));
520  acc_os_net_disconnect(sock);
521  return NULL;
522  }
523 
524  // use select() to detect activity on socket, or timeout
525  fd_set fdset;
526  FD_ZERO(&fdset);
527  FD_SET(sock->handle, &fdset);
528  struct timeval connect_timeout = { .tv_sec = 0, .tv_usec = 500000 };
529  int ret = select(sock->handle + 1, NULL, &fdset, NULL, &connect_timeout);
530  if (ret < 0) {
531  ACC_LOG_ERROR("%s: select() failed: (%u) %s", __func__, errno, strerror(errno));
532  acc_os_net_disconnect(sock);
533  return NULL;
534  }
535  if (!ret) {
536  char name[INET_ADDRSTRLEN];
537  acc_os_net_address_to_string(address, name, sizeof(name));
538  ACC_LOG_WARNING("%s: connect() to %s:%u timed out", __func__, name, port);
539  acc_os_net_disconnect(sock);
540  return NULL;
541  }
542 
543  // get any socket level errors
544  int so_error;
545  socklen_t len = sizeof(so_error);
546  if (getsockopt(sock->handle, SOL_SOCKET, SO_ERROR, &so_error, &len) < 0) {
547  ACC_LOG_ERROR("%s: getsockopt() failed: (%u) %s", __func__, errno, strerror(errno));
548  acc_os_net_disconnect(sock);
549  return NULL;
550  }
551 
552  // verify that connect() succeeded
553  if (so_error) {
554  char name[INET_ADDRSTRLEN];
555  acc_os_net_address_to_string(address, name, sizeof(name));
556  ACC_LOG_ERROR("%s: so_error for %s:%u: (%u) %s", __func__, name, port, so_error, strerror(so_error));
557  acc_os_net_disconnect(sock);
558  return NULL;
559  }
560 
561  // restore socket to blocking mode
562  fcntl(sock->handle, F_SETFL, sock_flags & ~O_NONBLOCK);
563 
564  int value = 1;
565  if (setsockopt(sock->handle, IPPROTO_TCP, TCP_NODELAY, (void*)&value, sizeof(value)) < 0) {
566  ACC_LOG_WARNING("%s: message_driver_tcp: setsockopt(TCP_NODELAY): (%u) %s", __func__, errno, strerror(errno));
567  acc_os_net_disconnect(sock);
568  return NULL;
569  }
570 
571  struct timeval send_timeout = { .tv_sec = 0, .tv_usec = 500000 };
572  if (setsockopt(sock->handle, SOL_SOCKET, SO_SNDTIMEO, (void*)&send_timeout, sizeof(send_timeout)) < 0)
573  ACC_LOG_WARNING("%s: setsockopt(SO_SNDTIMEO) failed: (%u) %s", __func__, errno, strerror(errno));
574 
575  return sock;
576 }

607 {
608  ssize_t result;
609  ssize_t remain = size;
610 
611  if (sock == NULL) {
612  return -1;
613  }
614 
615  while (remain) {
616  while (((result = send(sock->handle, buffer, remain, MSG_NOSIGNAL)) < 0) && (errno == EINTR)) ;
617  if (result < 0) {
618  ACC_LOG_ERROR("%s(): (%u) %s", __func__, errno, strerror(errno));
619  return -1;
620  }
621 
622  if (!result) {
623  ACC_LOG_ERROR("%s: send() sent zero bytes!", __func__);
624  return -1;
625  }
626 
627  buffer = (void*)((uint8_t*)buffer + result);
628  remain -= result;
629  }
630 
631  return size;

636 {

638  ssize_t remain = max_size;
639  size_t size = 0;
640 
641  if (sock == NULL) {
642  return -1;
643  }
644 
645  struct timeval receive_timeout = { .tv_sec = timeout_us / 1000000UL, .tv_usec = timeout_us % 1000000UL };
646  if (setsockopt(sock->handle, SOL_SOCKET, SO_RCVTIMEO, (void*)&receive_timeout, sizeof(receive_timeout)) < 0)

648 
649  while (remain) {

651 
652  if (result < 0) {
653  ACC_LOG_ERROR("%s: (%u) %s", __func__, errno, strerror(errno));

655  }
656 
657  // socket was closed by other end
658  if (!result) {
659  ACC_LOG_INFO("%s: Remote node closed connection", __func__);
660  return -1;
661  }
662 
663  buffer = (void*)((uint8_t*)buffer + result);
664  remain -= result;
665  size += result;
666  }
667 
668  return size;
669 }

689 int_fast8_t acc_os_semaphore_wait(acc_os_semaphore_t sem, uint_fast16_t timeout_ms)
690 {
691  struct timespec ts;
692  struct timeval tv;
693 
694  if (sem == NULL || sem->is_initialized != 1) {
695  ACC_LOG_ERROR("Not valid semaphore");
696  return -1;
697  }
698 
699  int result = gettimeofday(&tv, NULL);
700  if (result != 0) {
701  ACC_LOG_ERROR("gettimeofday returned %d %d %s", result, errno, strerror(errno));
702  return -1;
703  }
704 
705  ts.tv_sec = tv.tv_sec;
706  ts.tv_nsec = tv.tv_usec + (timeout_ms * 1000);
707 
708  while (ts.tv_nsec >= 1000000) {
709  ts.tv_sec++;
710  ts.tv_nsec -= 1000000;
711  }
712  ts.tv_nsec *= 1000;
713 
714  if (sem_timedwait(&(sem->handle), &ts) == -1) {
715  /* Timeout */
716  if (errno == ETIMEDOUT) {
717  if (timeout_ms != 0) {
718  ACC_LOG_ERROR("Semaphore timeout");
719  }
720  return -1;
721  }
722  ACC_LOG_ERROR("An error has occured: %d", errno);
723  return -1;
724  }
725 
726  return 0;
727 }
