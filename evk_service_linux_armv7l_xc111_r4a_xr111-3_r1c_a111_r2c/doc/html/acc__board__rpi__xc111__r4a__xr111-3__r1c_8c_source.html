135 {
136  acc_status_t status;

138  static acc_os_mutex_t init_mutex = NULL;
139 
140  if (init_done) {

144  acc_os_init();

146 
147  acc_os_mutex_lock(init_mutex);

151  }
152

154  (status = acc_device_gpio_set_initial_pull(GPIO0_PIN, 0)) ||
155  (status = acc_device_gpio_set_initial_pull(GPIO1_PIN, 0)) ||

160  (status = acc_device_gpio_set_initial_pull(PMU_ENABLE_PIN, 0)) ||
161  (status = acc_device_gpio_set_initial_pull(GLOREF_EN_PIN, 0)) ||
162  (status = acc_device_gpio_set_initial_pull(CE_A_PIN, 0)) ||
163  (status = acc_device_gpio_set_initial_pull(CE_B_PIN, 0))

166  }

168  if (

170  (status = acc_device_gpio_input(GPIO1_PIN)) ||
171  (status = acc_device_gpio_input(GPIO2_PIN)) ||
172  (status = acc_device_gpio_input(GPIO3_PIN)) ||
173  (status = acc_device_gpio_write(RSTn_PIN, 0)) ||
174  (status = acc_device_gpio_write(ENABLE_PIN, 0)) ||

177  (status = acc_device_gpio_write(CE_A_PIN, 0)) ||
178  (status = acc_device_gpio_write(CE_B_PIN, 0))
179  ) {
180  ACC_LOG_ERROR("%s failed with %s", __func__, acc_log_status_name(status));
181  acc_os_mutex_unlock(init_mutex);
182  return status;
183  }
184 
185  init_done = true;
186  acc_os_mutex_unlock(init_mutex);
187 
188  return ACC_STATUS_SUCCESS;
189 }

231 static acc_status_t acc_board_reset_sensor(void)
232 {

234 
235  status = acc_device_gpio_write(RSTn_PIN, 0);
236  if (status != ACC_STATUS_SUCCESS) {
237  ACC_LOG_ERROR("Unable to activate RSTn");

244  return status;
245  }

247  status = acc_device_gpio_write(PMU_ENABLE_PIN, 0);
248  if (status != ACC_STATUS_SUCCESS) {
249  ACC_LOG_ERROR("Unable to deactivate PMU_ENABLE");
250  return status;
251  }
252 
253  return ACC_STATUS_SUCCESS;
254 }

257 acc_status_t acc_board_start_sensor(acc_sensor_t sensor)
258 {
259  acc_status_t status;
260 
261  if (sensor_state[sensor - 1] == SENSOR_STATE_BUSY) {
262  ACC_LOG_ERROR("Sensor %u already active.", sensor);
263  return ACC_STATUS_FAILURE;
264  }
265 
266  if (acc_board_all_sensors_inactive()) {
267  status = acc_device_gpio_write(RSTn_PIN, 0);
268  if (status != ACC_STATUS_SUCCESS) {
269  ACC_LOG_ERROR("Unable to activate RSTn");
270  acc_board_reset_sensor();
271  return status;
272  }
273 
274  status = acc_device_gpio_write(PMU_ENABLE_PIN, 1);
275  if (status != ACC_STATUS_SUCCESS) {
276  ACC_LOG_ERROR("Unable to activate PMU_ENABLE");
277  acc_board_reset_sensor();
278  return status;
279  }
280 
281  // Wait for PMU to stabilize
282  acc_os_sleep_us(5000);
283 
284  status = acc_device_gpio_write(ENABLE_PIN, 1);
285  if (status != ACC_STATUS_SUCCESS) {
286  ACC_LOG_ERROR("Unable to activate ENABLE");
287  acc_board_reset_sensor();
288  return status;
289  }
290 
291  // Wait for Power On Reset
292  acc_os_sleep_us(5000);
293 
294  status = acc_device_gpio_write(RSTn_PIN, 1);
295  if (status != ACC_STATUS_SUCCESS) {
296  ACC_LOG_ERROR("Unable to deactivate RSTn");
297  acc_board_reset_sensor();
298  return status;
299  }
300 
301  for (uint_fast8_t sensor_index = 0; sensor_index < SENSOR_COUNT; sensor_index++) {
302  sensor_state[sensor_index] = SENSOR_STATE_READY;
303  }
304  }
305 
306  if (sensor_state[sensor - 1] != SENSOR_STATE_READY) {
307  ACC_LOG_ERROR("Sensor has not been reset");
308  return ACC_STATUS_FAILURE;
309  }
310 
311  sensor_state[sensor - 1] = SENSOR_STATE_BUSY;
312 
313  return ACC_STATUS_SUCCESS;
314 }

317 acc_status_t acc_board_stop_sensor(acc_sensor_t sensor)
318 {
319  if (sensor_state[sensor - 1] != SENSOR_STATE_BUSY) {
320  ACC_LOG_ERROR("Sensor %u already inactive.", sensor);
321  return ACC_STATUS_FAILURE;
322  }
323 
324  sensor_state[sensor - 1] = SENSOR_STATE_UNKNOWN;
325 
326  if (acc_board_all_sensors_inactive()) {
327  return acc_board_reset_sensor();

329 
330  return ACC_STATUS_SUCCESS;
