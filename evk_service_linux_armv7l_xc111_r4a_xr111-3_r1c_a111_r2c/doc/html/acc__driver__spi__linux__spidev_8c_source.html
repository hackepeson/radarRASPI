59 /**
60  * @brief Internal SPI open
61  *
62  * Open fd to /dev/spidevN.N.
63  *
64  * @param bus SPI bus
65  * @param device SPI device on bus
66  * @return Status
67  */
68 static acc_status_t internal_spi_open(uint_fast8_t bus, uint_fast8_t device)
69 {
70  uint32_t mode = 0;
71  char spidev[sizeof(SPIDEV_PATH) + 1];
72 
73  snprintf(spidev, sizeof(spidev), SPIDEV_PATH, bus, device);
74 
75  if ((spidev_fd[bus][device] = open(spidev, O_RDWR)) < 0) {
76  ACC_LOG_FATAL("Unable to open SPI (%u, %u): %s", bus, device, strerror(errno));
77  return ACC_STATUS_FAILURE;
78  }
79 
80  if (ioctl(spidev_fd[bus][device], _IOR('k', 1, uint8_t), &mode) < 0) {
81  ACC_LOG_WARNING("Could not set SPI (read) mode %u", mode);
82  }
83  if (ioctl(spidev_fd[bus][device], _IOW('k', 1, uint8_t), &mode) < 0) {
84  ACC_LOG_WARNING("Could not set SPI (write) mode %u", mode);
85  }
86 
87  return ACC_STATUS_SUCCESS;
88 }
